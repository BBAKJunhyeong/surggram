## Spacing 목록

컴포넌트 간 간격을 조정할 때 사용합니다. 값은 `number` 타입입니다.

<div
  style={{
    display: "flex",
    flexDirection: "column",
    gap: spacing.m,
  }}
>
  <div
    style={{
      backgroundColor: backgroundSecondary,
      display: "flex",
      justifyContent: "center",
      alignItems: "baseline",
      gap: spacing.xxxl,
      paddingTop: spacing.xxxl * 2,
      paddingBottom: spacing.xxxl * 2,
    }}
  >
    {Object.entries(spacing).map(([mainKey, spacingItem]) => (
      <div
        style={{
          height: 100,
          width: 50,
        }}
      >
        <div>
          <div
            key={mainKey}
            style={{
              width: spacingItem,
              height: spacingItem,
              backgroundColor: green400,
              marginBottom: spacing.m,
            }}
          ></div>
        </div>
        <div>
          <h3
            style={{
              marginBottom: spacing.xs,
            }}
          >{`${mainKey}`}</h3>
          <div
            style={{
              fontSize: 14,
              marginBottom: spacing.xs,
            }}
          >{`${spacingItem}`}</div>
        </div>
      </div>
    ))}
  </div>
</div>
